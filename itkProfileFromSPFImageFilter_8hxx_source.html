109  this->m_SPFToFeatureTransform = MatrixPointer (new MatrixType(n_b_sh,n_b) );
110  this->m_SPFToFeatureTransform->Fill(0.0);

113  {
114  spf->SetSPFType(SPFGenerator::SPF);
